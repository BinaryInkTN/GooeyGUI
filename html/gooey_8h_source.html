697void GooeyTextbox_HandleKeyPress(GooeyWindow *win, XKeyEvent *key_event)
698{
699 char buf[32];
700 KeySym key;
701 XLookupString(key_event, buf, sizeof(buf), &key, NULL);
702
703 for (int i = 0; i < win->textboxes_count; i++)
704 {
705 GooeyTextbox *textbox = &win->textboxes[i];
706 if (!textbox->focused)
707 continue;
708
709 size_t len = strlen(textbox->text);
710
711 if (key == XK_BackSpace)
712 {
713
714 if (len > 0)
715 {
716 textbox->text[len - 1] = '\0';
717 if (textbox->scroll_offset > 0)
718 {
719
720 textbox->scroll_offset--;
721 }
722
723 if (textbox->callback)
724 textbox->callback(textbox->text);
725 }
726 }
727 else if (key == XK_Return || key == XK_Linefeed)
728 {
729
730 textbox->focused = false;
731 }
732 else if (isprint(buf[0]) && len < sizeof(textbox->text) - 1)
733 {
734
735 strncat(textbox->text, buf, 1);
736
737 if (textbox->callback)
738 textbox->callback(textbox->text);
739
740 int text_width = XTextWidth(win->font, textbox->text, len + 1);
741 int max_text_width = textbox->core.width - 10;
742
743 if (text_width > max_text_width)
744 {
745 textbox->scroll_offset++;
746 }
747 }
748 }

909void GooeyLayout_Build(GooeyLayout *layout)
910{
911 if (!layout)
912 {
913 fprintf(stderr, "Error: Invalid layout pointer.\n");
914 return;
915 }
916
917 if (layout->widget_count == 0)
918 {
919 fprintf(stderr, "Error: Layout has no widgets to arrange.\n");
920 return;
921 }
922
923 int spacing = 30;
924 int current_x = layout->x;
925 int current_y = layout->y;
926
927 float max_widget_width = layout->layout_type == LAYOUT_HORIZONTAL ? layout->width / layout->widget_count : layout->width;
928
929 switch (layout->layout_type)
930 {
931 case LAYOUT_VERTICAL:
932 for (int i = 0; i < layout->widget_count; i++)
933 {
934 GooeyWidget *widget = layout->widgets[i];
935 widget->width = max_widget_width;
936 widget->x = layout->x + (layout->width - widget->width) / 2;
937 widget->y = current_y;
938 current_y += widget->height + spacing;
939 }
940 break;
941
942 case LAYOUT_HORIZONTAL:
943 for (int i = 0; i < layout->widget_count; i++)
944 {
945 GooeyWidget *widget = layout->widgets[i];
946 widget->width = max_widget_width;
947 widget->x = current_x;
948 widget->y = layout->y + (layout->height - widget->height) / 2;
949 current_x += widget->width + spacing;
950 }
951 break;
952
953 case LAYOUT_GRID:
954 {
955 break;
956 }
957
958 default:
959 fprintf(stderr, "Error: Unsupported layout type.\n");
960 break;
961 }
962}

1452 else if (GooeySlider_HandleDrag(win, x, y))
1453 {
1454 GooeyWindow_Redraw(win);
1455 }
1456 }
1457 else if (event.type == MotionNotify)

1461 else if (event.type == KeyPress)
