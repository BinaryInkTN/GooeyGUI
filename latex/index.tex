\chapter{Gooey GUI Library}
\hypertarget{index}{}\label{index}\index{Gooey GUI Library@{Gooey GUI Library}}
\label{index_md_gooey__homepage}%
\Hypertarget{index_md_gooey__homepage}%
 Gooey is a lightweight and feature-\/rich GUI (Graphical User Interface) library designed for X11 environments. It provides a range of customizable widgets and layouts for creating responsive and modern desktop applications. The library is optimized for simplicity and offers a flexible API for developers.\hypertarget{index_autotoc_md1}{}\doxysection{\texorpdfstring{Features}{Features}}\label{index_autotoc_md1}

\begin{DoxyItemize}
\item {\bfseries{Wide Range of Widgets}}\+:
\begin{DoxyItemize}
\item Labels
\item Buttons
\item Checkboxes
\item Radio Buttons (and Groups)
\item Sliders
\item Dropdown Menus
\item Textboxes
\end{DoxyItemize}
\item {\bfseries{Layouts}}\+:
\begin{DoxyItemize}
\item Horizontal
\item Vertical
\item Grid-\/based layouts
\end{DoxyItemize}
\item {\bfseries{Customization Options}}\+:
\begin{DoxyItemize}
\item Support for widget-\/specific callbacks
\item Styling with predefined color schemes
\item Interactive hover and click states
\end{DoxyItemize}
\item {\bfseries{Event Handling}}\+:
\begin{DoxyItemize}
\item Mouse clicks, motion, and keypress events
\item Dynamic updates through an efficient redraw system
\end{DoxyItemize}
\end{DoxyItemize}\hypertarget{index_autotoc_md2}{}\doxysection{\texorpdfstring{Getting Started}{Getting Started}}\label{index_autotoc_md2}
