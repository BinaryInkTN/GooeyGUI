547{
548 GooeyWindow win;
549 win.display = XOpenDisplay(NULL);
550 if (!win.display)
551 {
552 fprintf(stderr, "Unable to open X display\n");
553 exit(1);
554 }
555 win.font = XLoadQueryFont(win.display, "fixed");
556 if (!win.font)
557 {
558 fprintf(stderr, "Failed to load system font\n");
559 exit(1);
560 }
561
562 int screen = DefaultScreen(win.display);
563 win.window = XCreateSimpleWindow(
564 win.display, RootWindow(win.display, screen), 10, 10, width, height, 1,
565 BlackPixel(win.display, screen), WhitePixel(win.display, screen));
566 XStoreName(win.display, win.window, title);
567
568 win.gc = XCreateGC(win.display, win.window, 0, NULL);
569 win.wm_delete_window = XInternAtom(win.display, "WM_DELETE_WINDOW", False);
570 XSetWMProtocols(win.display, win.window, &win.wm_delete_window, 1);
571
572 XSelectInput(win.display, win.window,
573 ExposureMask | ButtonPressMask | ButtonMotionMask |
574 KeyPressMask);
575 XMapWindow(win.display, win.window);
576
577 win.button_count = 0;
578 win.label_count = 0;
579 win.checkbox_count = 0;
580 win.radio_button_count = 0;
581 win.radio_button_group_count = 0;
582 win.slider_count = 0;
583 win.dropdown_count = 0;
584 win.textboxes_count = 0;
585 win.layout_count = 0;

676void GooeyTextbox_Draw(GooeyWindow *win, int index)
677{
678 GooeyTextbox *textbox = &win->textboxes[index];
679
680 XSetForeground(win->display, win->gc, COLOR_WHITE);
681 XFillRectangle(win->display, win->window, win->gc, textbox->core.x, textbox->core.y,
682 textbox->core.width, textbox->core.height);
683
684 XSetForeground(win->display, win->gc,
685 textbox->focused ? COLOR_BLUE : COLOR_BLACK);
686 XDrawRectangle(win->display, win->window, win->gc, textbox->core.x, textbox->core.y,
687 textbox->core.width, textbox->core.height);
688
689 XSetForeground(win->display, win->gc, COLOR_BLACK);
690 int text_x = textbox->core.x + 5;
691 int text_y = textbox->core.y + (textbox->core.height / 2) + 5;
692
693 int max_text_width = textbox->core.width - 10;
694 size_t len = strlen(textbox->text);
695 size_t start_index = textbox->scroll_offset;
696
697 while (start_index < len && XTextWidth(win->font, textbox->text + start_index,
698 len - start_index) > max_text_width)
699 {
700 start_index++;
701 }
702
703 char display_text[256];
704 strncpy(display_text, textbox->text + start_index, sizeof(display_text) - 1);
705 display_text[sizeof(display_text) - 1] = '\0';
706
707 XDrawString(win->display, win->window, win->gc, text_x, text_y, display_text,
708 strlen(display_text));
709
710 if (textbox->focused)
711 {
712 int cursor_x =
713 text_x + XTextWidth(win->font, display_text, strlen(display_text));
714 XDrawLine(win->display, win->window, win->gc, cursor_x, textbox->core.y + 5,
715 cursor_x, textbox->core.y + textbox->core.height - 5);
716 }

718void GooeyTextbox_HandleKeyPress(GooeyWindow *win, XKeyEvent *key_event)
719{
720 char buf[32];
721 KeySym key;
722 XLookupString(key_event, buf, sizeof(buf), &key, NULL);
723
724 for (int i = 0; i < win->textboxes_count; i++)
725 {
726 GooeyTextbox *textbox = &win->textboxes[i];
727 if (!textbox->focused)
728 continue;
729
730 size_t len = strlen(textbox->text);
731
732 if (key == XK_BackSpace)
733 {
734
735 if (len > 0)
736 {
737 textbox->text[len - 1] = '\0';
738 if (textbox->scroll_offset > 0)
739 {
740
741 textbox->scroll_offset--;
742 }
743
744 if (textbox->callback)
745 textbox->callback(textbox->text);
746 }
747 }
748 else if (key == XK_Return || key == XK_Linefeed)
749 {
750
751 textbox->focused = false;
752 }
753 else if (isprint(buf[0]) && len < sizeof(textbox->text) - 1)
754 {
755
756 strncat(textbox->text, buf, 1);
757
758 if (textbox->callback)
759 textbox->callback(textbox->text);
760
761 int text_width = XTextWidth(win->font, textbox->text, len + 1);
762 int max_text_width = textbox->core.width - 10;
763
764 if (text_width > max_text_width)
765 {
766 textbox->scroll_offset++;
767 }
768 }
769 }

930void GooeyLayout_Build(GooeyLayout *layout)
931{
932 if (!layout)
933 {
934 fprintf(stderr, "Error: Invalid layout pointer.\n");
935 return;
936 }
937
938 if (layout->widget_count == 0)
939 {
940 fprintf(stderr, "Error: Layout has no widgets to arrange.\n");
941 return;
942 }
943
944 int spacing = 30;
945 int current_x = layout->x;
946 int current_y = layout->y;
947
948 float max_widget_width = layout->layout_type == LAYOUT_HORIZONTAL ? layout->width / layout->widget_count : layout->width;
949
950 switch (layout->layout_type)
951 {
952 case LAYOUT_VERTICAL:
953 for (int i = 0; i < layout->widget_count; i++)
954 {
955 GooeyWidget *widget = layout->widgets[i];
956 widget->width = max_widget_width;
957 widget->x = layout->x + (layout->width - widget->width) / 2;
958 widget->y = current_y;
959 current_y += widget->height + spacing;
960 }
961 break;
962
963 case LAYOUT_HORIZONTAL:
964 for (int i = 0; i < layout->widget_count; i++)
965 {
966 GooeyWidget *widget = layout->widgets[i];
967 widget->width = max_widget_width;
968 widget->x = current_x;
969 widget->y = layout->y + (layout->height - widget->height) / 2;
970 current_x += widget->width + spacing;
971 }
972 break;
973
974 case LAYOUT_GRID:
975 {
976 break;
977 }
978
979 default:
980 fprintf(stderr, "Error: Unsupported layout type.\n");
981 break;
982 }
983}
